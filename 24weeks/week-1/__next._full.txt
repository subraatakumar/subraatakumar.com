import { createOpenAI } from '@ai-sdk/openai';
import {
  convertToModelMessages,
  streamText,
  type UIMessage,
  type LanguageModel,
} from 'ai';

type ProviderName = 'google' | 'openai' | 'anthropic';

type PanelConfig = {
  provider: ProviderName;
  modelId: string;
  apiKey?: string;
};

type ChatRequestBody = {
  messages: UIMessage[];
  panel: PanelConfig;
  options?: {
    systemPrompt?: string;
    temperature?: number;
  };
};

function getModel(panel: PanelConfig): LanguageModel {
  const modelId = panel.modelId.trim();
  const apiKey = panel.apiKey?.trim();

  if (!modelId) throw new Error('Model ID is required.');

  if (panel.provider === 'google') {
    const fallbackKey = process.env.GOOGLE_API_KEY?.trim();
    const googleApiKey = apiKey || fallbackKey;
    if (!googleApiKey) {
      throw new Error('No Google API key found. Add GOOGLE_API_KEY or provide panel key.');
    }
    const google = createGoogleGenerativeAI({ apiKey: googleApiKey });
    return google(modelId);
  }

  if (!apiKey) {
    throw new Error(`API key is required for ${panel.provider}.`);
  }

  if (panel.provider === 'openai') {
    const openai = createOpenAI({ apiKey });
    return openai(modelId);
  }

  const anthropic = createAnthropic({ apiKey });
  return anthropic(modelId);
}

export async function POST(req: Request) {
  const body = (await req.json()) as ChatRequestBody;
  const { messages, panel, options } = body;

  const model = getModel(panel);

  const result = streamText({
    model,
    system: options?.systemPrompt?.trim() || undefined,
    temperature: options?.temperature,
    messages: await convertToModelMessages(messages),
  });

  return result.toUIMessageStreamResponse({ originalMessages: messages });
}

<li class="d180-md-li">Returning non-UI stream response formats with <code class="d180-md-inline-code">useChat</code>.</li>
</ul>

<h2 id="comparison-report-logic-handles-partial-failure" class="d180-md-h2"><strong>Comparison Report Logic (Handles Partial Failure)</strong></h2>
<p class="d180-md-p">A key Week 1 learning: comparison should still be useful if one side fails.</p>

      <div class="relative my-6 code-block">
        <div class="absolute right-3 top-3 flex items-center gap-2 z-50 pointer-events-auto">
          <span class="d180-code-lang">TS</span>
          <button data-copy class="copy-btn d180-copy-btn">Copy</button>
        </div>
        <pre class="d180-code-pre line-numbers"><code class="language-ts">function getComparisonReport({ leftText, rightText, leftError, rightError }) {
  const leftOk = Boolean(leftText) &#x26;&#x26; !leftError;
  const rightOk = Boolean(rightText) &#x26;&#x26; !rightError;

  if (leftOk &#x26;&#x26; rightOk) {
    return { summary: 'Both models returned successfully.', winner: '...'};
  }

  if (leftOk &#x26;&#x26; !rightOk) {
    return { summary: 'Partial success: left model succeeded, right model failed.', winner: 'Left model by availability' };
  }

  if (!leftOk &#x26;&#x26; rightOk) {
    return { summary: 'Partial success: right model succeeded, left model failed.', winner: 'Right model by availability' };
  }

  return { summary: 'Both model runs failed.', winner: 'No winner' };
}
</code></pre>
      </div>
    
<h3 id="why-this-matters" class="d180-md-h3">Why this matters</h3>
<p class="d180-md-p">Without this, users lose confidence in the evaluator when one provider fails.
With this, users still get actionable output and next-step guidance.</p>

<h2 id="what-broke-during-build-and-how-we-fixed-it" class="d180-md-h2"><strong>What Broke During Build (and How We Fixed It)</strong></h2>
<h3 id="1-old-usechat-api-body-pattern-mismatch" class="d180-md-h3">1) Old <code class="d180-md-inline-code">useChat({ api, body })</code> pattern mismatch</h3>
<p class="d180-md-p"><strong>Problem:</strong> Newer AI SDK uses transport objects.</p>
<p class="d180-md-p"><strong>Fix:</strong> Move to <code class="d180-md-inline-code">DefaultChatTransport</code> and pass per-request options with <code class="d180-md-inline-code">sendMessage(..., { body })</code>.</p>
<h3 id="2-old-messagecontent-rendering" class="d180-md-h3">2) Old <code class="d180-md-inline-code">message.content</code> rendering</h3>
<p class="d180-md-p"><strong>Problem:</strong> Modern UI messages expose <code class="d180-md-inline-code">parts</code>, not single <code class="d180-md-inline-code">content</code>.</p>
<p class="d180-md-p"><strong>Fix:</strong> Render assistant text by collecting text parts from <code class="d180-md-inline-code">message.parts</code>.</p>
<h3 id="3-stuck-running-status-after-provider-error" class="d180-md-h3">3) Stuck running status after provider error</h3>
<p class="d180-md-p"><strong>Problem:</strong> UI considered non-ready statuses as running.</p>
<p class="d180-md-p"><strong>Fix:</strong> Busy only for <code class="d180-md-inline-code">submitted</code> and <code class="d180-md-inline-code">streaming</code>.</p>
<h3 id="4-no-comparison-when-one-model-failed" class="d180-md-h3">4) No comparison when one model failed</h3>
<p class="d180-md-p"><strong>Problem:</strong> Successful side had output, but no final cross-panel summary.</p>
<p class="d180-md-p"><strong>Fix:</strong> Add partial-failure report path.</p>
<h3 id="5-quotarate-limit-confusion" class="d180-md-h3">5) Quota/rate-limit confusion</h3>
<p class="d180-md-p"><strong>Problem:</strong> Users thought app was broken when provider quota returned <code class="d180-md-inline-code">limit: 0</code>.</p>
<p class="d180-md-p"><strong>Fix:</strong> Surface raw provider error in panel + add quota troubleshooting guidance.</p>

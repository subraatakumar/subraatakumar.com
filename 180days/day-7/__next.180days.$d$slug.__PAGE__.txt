import { v4 as uuidv4 } from 'uuid';

export type CreateEntryInput = {
  title: string;
  content: string;
};

export class ArchiveService {
  private repository = new ArchiveRepository();

  createEntry(input: CreateEntryInput) {
    if (!input.title.trim()) {
      throw new Error('Title cannot be empty');
    }

    const entry = {
      id: uuidv4(),
      title: input.title.trim(),
      content: input.content.trim(),
      createdAt: new Date(),
    };

    this.repository.create(entry);
  }

  getEntries() {
    return this.repository.getAll();
  }

  deleteEntry(id: string) {
    this.repository.delete(id);
  }
}

<h3 id="3-move-uuid-to-repository-then-undo-it" class="d180-md-h3">3️⃣ Move UUID to Repository (Then Undo It)</h3>
<p class="d180-md-p">Temporarily shift ID generation into repository.</p>
<p class="d180-md-p">Observe how it feels:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">Repository now decides identity.</li>
<li class="d180-md-li">Business logic leaks into data layer.</li>
<li class="d180-md-li">Responsibility becomes unclear.</li>
</ul>
<p class="d180-md-p">Undo it.</p>
<p class="d180-md-p">Identity belongs to service.</p>
<p class="d180-md-p">Repository should not “decide.”</p>
<hr class="d180-md-hr" />

<h3 id="5-try-calling-repository-directly-from-ui-then-delete-it" class="d180-md-h3">5️⃣ Try Calling Repository Directly from UI (Then Delete It)</h3>

<h3 id="-what-day-7-really-teaches" class="d180-md-h3">🧠 What Day 7 Really Teaches</h3>
<p class="d180-md-p">Repository protects access.</p>
<p class="d180-md-p">Service protects logic.</p>
<p class="d180-md-p">Encryption protects data.</p>
<p class="d180-md-p">Each layer has one responsibility.</p>
<p class="d180-md-p">And that discipline is what protects the system long-term .</p>

<h3 id="-service-layer-pattern-conceptual-foundation" class="d180-md-h3">🧠 Service Layer Pattern (Conceptual Foundation)</h3>
<p class="d180-md-p">Martin Fowler — Service Layer Pattern:</p>
<p class="d180-md-p"><a href="https://martinfowler.com/eaaCatalog/serviceLayer.html" class="d180-md-link">https://martinfowler.com/eaaCatalog/serviceLayer.html</a></p>
<p class="d180-md-p">This explains:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">Why business logic should not live inside controllers (or UI)</li>
<li class="d180-md-li">How services coordinate workflows</li>
<li class="d180-md-li">Why separation prevents chaos as systems grow</li>
</ul>
<p class="d180-md-p">Today’s implementation directly aligns with this idea.</p>
<hr class="d180-md-hr" />
<h3 id="-clean-architecture-layer-responsibility" class="d180-md-h3">🧱 Clean Architecture (Layer Responsibility)</h3>
<p class="d180-md-p">Robert C. Martin (Uncle Bob) — Clean Architecture Concepts:</p>
<p class="d180-md-p"><a href="https://blog.cleancoder.com/" class="d180-md-link">https://blog.cleancoder.com/</a></p>
<p class="d180-md-p">Focus on:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">Separation of concerns</li>
<li class="d180-md-li">Dependency direction</li>
<li class="d180-md-li">Keeping business rules independent of frameworks</li>
</ul>
<p class="d180-md-p">Our layering model reflects this philosophy .</p>
<hr class="d180-md-hr" />
<h3 id="-repository-vs-service-clear-responsibility-difference" class="d180-md-h3">🏗 Repository vs Service (Clear Responsibility Difference)</h3>
<p class="d180-md-p">A helpful explanation of responsibilities:</p>
<p class="d180-md-p"><a href="https://enterprisecraftsmanship.com/posts/service-layer-vs-repository-pattern/" class="d180-md-link">https://enterprisecraftsmanship.com/posts/service-layer-vs-repository-pattern/</a></p>
<p class="d180-md-p">Understand:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">Repository = data access abstraction</li>
<li class="d180-md-li">Service = business rule orchestration</li>
<li class="d180-md-li">Why mixing them creates long-term instability</li>
</ul>

<p class="d180-md-p">Official UUID package documentation:</p>
<p class="d180-md-p"><a href="https://www.npmjs.com/package/uuid" class="d180-md-link">https://www.npmjs.com/package/uuid</a></p>
<p class="d180-md-p">Why this matters:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">ID generation belongs to business layer</li>
<li class="d180-md-li">Identity is not a storage concern</li>
<li class="d180-md-li">Separation of identity from persistence keeps system flexible</li>
</ul>

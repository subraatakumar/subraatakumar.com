import { fromByteArray, toByteArray } from 'react-native-quick-base64';

const ENCRYPTION_KEY_ALIAS = 'offline_life_archive_key';

export async function storeEncryptionKey(key: Uint8Array) {
  const base64Key = fromByteArray(key);

  await Keychain.setGenericPassword(
    ENCRYPTION_KEY_ALIAS,
    base64Key,
    {
      service: ENCRYPTION_KEY_ALIAS,
      accessible: Keychain.ACCESSIBLE.WHEN_UNLOCKED_THIS_DEVICE_ONLY,
    }
  );
}

export async function getEncryptionKey(): Promise&#x3C;Uint8Array | null> {
  const credentials = await Keychain.getGenericPassword({
    service: ENCRYPTION_KEY_ALIAS,
  });

  if (!credentials) return null;

  return toByteArray(credentials.password);
}

let realmInstance: Realm | null = null;

export async function initializeRealm(): Promise&#x3C;Realm> {
  if (realmInstance) {
    return realmInstance;
  }

  let key = await getEncryptionKey();

  if (!key) {
    key = generateEncryptionKey();
    await storeEncryptionKey(key);
  }

  realmInstance = await Realm.open({
    schema: schemas,
    encryptionKey: key,
  });

  return realmInstance;
}

export function getRealm(): Realm {
  if (!realmInstance) {
    throw new Error('Realm has not been initialized.');

<li class="d180-md-li">New encryption key generated.</li>
<li class="d180-md-li">Fresh Realm database created.</li>
<li class="d180-md-li">No crash.</li>
</ul>
<p class="d180-md-p">Because Keychain entry is removed when app is deleted.</p>
<hr class="d180-md-hr" />
<h3 id="-5-verify-no-plaintext-database" class="d180-md-h3">✅ 5️⃣ Verify No Plaintext Database</h3>
<p class="d180-md-p">On Android:</p>
<p class="d180-md-p">Navigate to:</p>

      <div class="relative my-6 code-block">
        <div class="absolute right-3 top-3 flex items-center gap-2 z-50 pointer-events-auto">
          
          <button data-copy class="copy-btn d180-copy-btn">Copy</button>
        </div>
        <pre class="d180-code-pre line-numbers"><code>/data/data/&#x3C;your.package.name>/files/
</code></pre>
      </div>
    
<p class="d180-md-p">Pull the Realm file.</p>
<p class="d180-md-p">Open it with a text editor or hex viewer.</p>
<p class="d180-md-p">Expected:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">You should NOT see readable strings like entry titles.</li>
<li class="d180-md-li">Data should appear binary / unreadable.</li>
</ul>
<p class="d180-md-p">If you see readable text:</p>
<p class="d180-md-p">🚨 Encryption is not applied.</p>
<hr class="d180-md-hr" />
<h3 id="-6-ensure-key-is-never-logged-in-production" class="d180-md-h3">✅ 6️⃣ Ensure Key Is Never Logged in Production</h3>
<p class="d180-md-p">Search your codebase:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">Ensure encryption key is never console.logged.</li>
<li class="d180-md-li">Ensure no debug statements expose it.</li>
</ul>
<p class="d180-md-p">You may log it temporarily in <code class="d180-md-inline-code">__DEV__</code>, but remove before production.</p>
<hr class="d180-md-hr" />
<h3 id="-7-confirm-realm-file-exists" class="d180-md-h3">✅ 7️⃣ Confirm Realm File Exists</h3>
<p class="d180-md-p">Add temporary log:</p>

      <div class="relative my-6 code-block">
        <div class="absolute right-3 top-3 flex items-center gap-2 z-50 pointer-events-auto">
          <span class="d180-code-lang">TSX</span>
          <button data-copy class="copy-btn d180-copy-btn">Copy</button>
        </div>
        <pre class="d180-code-pre line-numbers"><code class="language-tsx">console.log(Realm.defaultPath);
</code></pre>
      </div>
    
<p class="d180-md-p">Confirm:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">File path exists.</li>
<li class="d180-md-li">Database file size increases after adding data (later).</li>
</ul>
<p class="d180-md-p">This confirms actual storage location.</p>
<hr class="d180-md-hr" />
<h1 id="-common-mistakes" class="d180-md-h1">🚨 Common Mistakes</h1>
<p class="d180-md-p">❌ Using AsyncStorage for encryption key</p>
<p class="d180-md-p">❌ Generating a new key every launch</p>
<p class="d180-md-p">❌ Storing key without base64 encoding</p>
<p class="d180-md-p">❌ Forgetting to pod install (iOS)</p>
<p class="d180-md-p">❌ Using incorrect key length (must be 64 bytes)</p>
<hr class="d180-md-hr" />
<h1 id="-final-confirmation-logic" class="d180-md-h1">🧠 Final Confirmation Logic</h1>
<p class="d180-md-p">If all these are true:</p>
<ul class="d180-md-ul">
<li class="d180-md-li">App runs on first launch</li>
<li class="d180-md-li">App runs on restart</li>
<li class="d180-md-li">App fails when key is changed</li>
<li class="d180-md-li">Database file unreadable externally</li>
</ul>
<p class="d180-md-p">Then:</p>
<p class="d180-md-p">✔ Encryption is correctly implemented</p>
<p class="d180-md-p">✔ Key management is stable</p>
<p class="d180-md-p">✔ Core layer is functioning</p>
<hr class="d180-md-hr" />
<p class="d180-md-p">This checklist converts encryption from theory → verified infrastructure.</p>

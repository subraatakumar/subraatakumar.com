<h2 id="52-why-encryption-from-day-one" class="text-2xl lg:text-3xl font-black tracking-tight text-slate-900 dark:text-white mt-14 mb-5">5.2 Why Encryption From Day One?</h2>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">From the product definition , this application is:</p>
<blockquote class="border-l-2 border-slate-200 dark:border-slate-800 bg-slate-50 dark:bg-slate-900/30 p-4 rounded-lg text-slate-700 dark:text-slate-300 italic">
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">“Your private digital life. Stored locally. Owned by you.”</p>
</blockquote>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">Privacy is not an optional enhancement.</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">It is the core value proposition.</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">If we delay encryption:</p>
<ul class="list-disc pl-6 space-y-1 my-4">

<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">We introduce refactor risk later.</li>
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">We compromise architectural integrity.</li>
</ul>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">Therefore, encryption is implemented before any feature logic.</p>

import { fromByteArray, toByteArray } from 'react-native-quick-base64';

const ENCRYPTION_KEY_ALIAS = 'offline_life_archive_key';

export async function storeEncryptionKey(key: Uint8Array) {
  const base64Key = fromByteArray(key);

  await Keychain.setGenericPassword(
    ENCRYPTION_KEY_ALIAS,
    base64Key,
    {
      service: ENCRYPTION_KEY_ALIAS,
      accessible: Keychain.ACCESSIBLE.WHEN_UNLOCKED_THIS_DEVICE_ONLY,
    }
  );
}

export async function getEncryptionKey(): Promise&#x3C;Uint8Array | null> {
  const credentials = await Keychain.getGenericPassword({
    service: ENCRYPTION_KEY_ALIAS,
  });

  if (!credentials) return null;

  return toByteArray(credentials.password);
}

import { generateEncryptionKey } from '../security/encryptionKey';
import { storeEncryptionKey, getEncryptionKey } from '../security/secureStorage';

let realmInstance: Realm | null = null;

export async function initializeRealm(): Promise&#x3C;Realm> {
  if (realmInstance) {
    return realmInstance;
  }

  let key = await getEncryptionKey();

  if (!key) {
    key = generateEncryptionKey();
    await storeEncryptionKey(key);
  }

  realmInstance = await Realm.open({
    schema: schemas,
    encryptionKey: key,
  });

  return realmInstance;
}

export function getRealm(): Realm {
  if (!realmInstance) {
    throw new Error('Realm has not been initialized.');
  }

  return realmInstance;

<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">Ensure no debug statements expose it.</li>
</ul>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">You may log it temporarily in <code>__DEV__</code>, but remove before production.</p>
<hr>
<h3 id="-7-confirm-realm-file-exists" class="text-xl lg:text-2xl font-bold tracking-tight text-slate-900 dark:text-white mt-10 mb-4">✅ 7️⃣ Confirm Realm File Exists</h3>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">Add temporary log:</p>

      <div class="relative my-6 code-block">
        <div class="absolute right-3 top-3 flex items-center gap-2 z-50 pointer-events-auto">
          <span class="text-[11px] font-medium text-slate-500 dark:text-slate-400 uppercase tracking-wider bg-slate-100 dark:bg-slate-800 px-2.5 py-1 rounded z-50">TSX</span>
          <button data-copy class="copy-btn text-[11px] font-medium px-2.5 py-1 rounded bg-amber-400 text-slate-900 hover:bg-amber-500 z-50 uppercase tracking-wider">Copy</button>
        </div>
        <pre class="overflow-auto rounded-2xl bg-slate-950 text-slate-100 p-4 line-numbers"><code class="language-tsx">console.log(Realm.defaultPath);
</code></pre>
      </div>
    
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">Confirm:</p>
<ul class="list-disc pl-6 space-y-1 my-4">
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">File path exists.</li>
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">Database file size increases after adding data (later).</li>
</ul>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">This confirms actual storage location.</p>

<h1 id="-common-mistakes" class="text-3xl lg:text-4xl font-black tracking-tighter text-slate-950 dark:text-white mt-16 mb-6 pb-4 border-b border-slate-200 dark:border-slate-700">🚨 Common Mistakes</h1>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">❌ Using AsyncStorage for encryption key</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">❌ Generating a new key every launch</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">❌ Storing key without base64 encoding</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">❌ Forgetting to pod install (iOS)</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">❌ Using incorrect key length (must be 64 bytes)</p>
<hr>
<h1 id="-final-confirmation-logic" class="text-3xl lg:text-4xl font-black tracking-tighter text-slate-950 dark:text-white mt-16 mb-6 pb-4 border-b border-slate-200 dark:border-slate-700">🧠 Final Confirmation Logic</h1>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">If all these are true:</p>
<ul class="list-disc pl-6 space-y-1 my-4">
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">App runs on first launch</li>
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">App runs on restart</li>
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">App fails when key is changed</li>
<li class="text-[17px] leading-relaxed text-slate-700 dark:text-slate-300 my-1.5">Database file unreadable externally</li>
</ul>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">Then:</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">✔ Encryption is correctly implemented</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">✔ Key management is stable</p>
<p class="text-[17px] leading-[1.85] text-slate-700 dark:text-slate-300 my-5">✔ Core layer is functioning</p>
